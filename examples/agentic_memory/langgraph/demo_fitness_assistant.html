The fitness assistant showcases long-term memory functionality, enabling it to remember user preferences, past conversations, and fitness history across multiple sessions.

Key FeaturesÔÉÅ

  • Semantic Search: Retrieves relevant memories based on context and meaning

  • Session Continuity: Maintains context even when conversations are resumed later

Step 4: Create ChatbotÔÉÅ

from langchain_aws import ChatBedrock
from langgraph.graph import StateGraph, MessagesState, START, END
from langgraph.prebuilt import ToolNode

def create_chatbot(checkpointer, memory_tools):
    """Create a minimal chatbot with checkpoint support."""

    model = ChatBedrock(
        model_id=bedrock_model_id,
        region_name=os.getenv('AWS_REGION'),
        model_kwargs={"temperature": 0.7, "max_tokens": 1024}
    )

    model_with_tools = model.bind_tools(memory_tools)

    def chat_node(state: MessagesState):
        messages = state["messages"]
        if not any(isinstance(msg, SystemMessage) for msg in messages):
            system_msg = SystemMessage(content="""You are a professional fitness coach with access to client memory.
            Always store important client information (fitness goals, workout preferences, injuries, restrictions, progress) using the opensearch_memory tool.
            When clients return, search your memory to provide personalized fitness recommendations and track their progress.
            Focus on safe, effective workouts tailored to each client's fitness level and limitations.""")
            messages = [system_msg] + messages
        return {"messages": [model_with_tools.invoke(messages)]}

    def should_continue(state: MessagesState):
        messages = state['messages']
        last_message = messages[-1]
        if last_message.tool_calls:
            return "tools"
        return END

    graph = StateGraph(MessagesState)
    graph.add_node("chat", chat_node)
    graph.add_node("tools", ToolNode(memory_tools))

    graph.add_edge(START, "chat")
    graph.add_conditional_edges("chat", should_continue)
    graph.add_edge("tools", "chat")

    return graph.compile(checkpointer=checkpointer)

print("Creating chatbot...")
app = create_chatbot(checkpointer, memory_provider.tools)
print("‚úÖ Chatbot ready")

from datetime import datetime
from langchain_core.messages import HumanMessage, SystemMessage

def find_existing_thread(container_id: str, user_prefix: str = "demo_") -> str:
    """Find the most recent thread ID for this user."""
    import requests

    auth = (opensearch_username, opensearch_password)
    url = f"{opensearch_url}/_plugins/_ml/memory_containers/{container_id}/memories/working/_search"

    body = {
        "query": {
            "bool": {
                "filter": [
                    {"term": {"payload_type": "data"}},
                    {"term": {"metadata.type": "checkpoint"}},
                    {"wildcard": {"namespace.thread_id": f"{user_prefix}*"}}
                ]
            }
        },
        "sort": [{"checkpoint_id": {"order": "desc"}}],
        "size": 1
    }

    try:
        response = requests.post(url, json=body, auth=auth, verify=opensearch_verify_ssl)
        response.raise_for_status()
        data = response.json()

        hits = data.get("hits", {}).get("hits", [])
        if hits:
            return hits[0]["_source"]["namespace"]["thread_id"]
        return None
    except Exception:
        return None

def send_message(app, thread_id: str, message: str):
    """Send a message and return the response."""
    config = {"configurable": {"thread_id": thread_id}}
    result = app.invoke({"messages": [HumanMessage(content=message)]}, config)
    return result["messages"][-1].content


def get_message_count(app, thread_id: str) -> int:
    """Get the number of messages in the conversation."""
    config = {"configurable": {"thread_id": thread_id}}
    state = app.get_state(config)
    if state and state.values:
        return len(state.values.get("messages", []))
    return 0

def start_fitness_session(session_name):
    """Start a fitness agent conversation session"""

    # Check for existing threads
    existing_thread = find_existing_thread(memory_provider.memory.memory_container_id)

    if existing_thread:
        print(f"Found existing thread: {existing_thread}")
        choice = input("Resume existing conversation? (y/n): ").strip().lower()
        if choice == 'y':
            thread_id = existing_thread
            print(f"Resuming thread: {thread_id}")
        else:
            thread_id = f"demo_{datetime.now().strftime('%Y%m%d_%H%M%S')}"
            print(f"Starting new thread: {thread_id}")
    else:
        thread_id = f"demo_{datetime.now().strftime('%Y%m%d_%H%M%S')}"
        print(f"Starting new thread: {thread_id}")

    print("üí™ Fitness Coach Interactive Demo")
    print("Type 'q' or 'quit' to end the conversation\n")

    while True:
        question = input("üë§ You: ").strip()

        if question.lower() in ['q', 'quit']:
            print("üí™ Fitness Agent: Keep up the great work! I'll remember our progress for next time.")

            # Show session summary
            msg_count = get_message_count(app, thread_id)
            print("\n" + "=" * 70)
            print(f"\nüìä Session Summary:")
            print(f"Thread ID: {thread_id}")
            print(f"Total messages: {msg_count}")
            print(f"Session ended: {datetime.now()}")
            break

        if not question:
            continue

        response = send_message(app, thread_id, question)
        print(f"üí™ Fitness Agent: {response}")
        print()
